	set_data_policy_open = { 
		potential = {
            always = no
            has_country_flag = show_institution_decisions
            NOT = { has_country_modifier = institution_policy_cooldown }
		}
		allow =  {
			NOT = { has_country_modifier = institution_policy_cooldown }
            NOT = { has_country_modifier = institution_data_breach }
		}
	  	effect = {
			remove_country_modifier = institution_access_open
            remove_country_modifier = institution_access_restricted
            remove_country_modifier = institution_access_closed
            
            add_country_modifier = {
                name = institution_access_open
                duration = -1
            }
            
            add_country_modifier = {
                name = institution_policy_cooldown
                duration = 365
                hidden = yes
            }
		}

			factor = 0
		}
	}
