estate_faith =
{
	icon = 6
	
	color = { 33 165 100 }
	
    independence_government = administrative_settlement
	independence_tech_group = post_apocalyptic
    independence_trigger = {
		has_owner_religion = no
	}
    
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no
	
	# They want to control at least this % of the country's non-overseas development, if # of home provinecs is above set amount
	min_territory = 0
	min_provinces_to_want_territory = 5
    
	# If true, country will get estate
	trigger = {	
        always = no
		NOT = { technology_group = tribal }
        NOT = { technology_group = tech_super_mutant }
        any_owned_province = {
            NOT = { religion = owner }
        }
        NOT = { tag = ZET }
	}
	
	# If true, province can be granted to estate
	province_trigger = {
        NOT = { religion = owner }
		NOT = { religion = secular }
	}	
	
	# Min autonomy in estate provinces
	min_autonomy = 25
	
	# Estate cancels out the following multiplicative local autonomy effects:
	autonomy_does_not_affect = local_tax_modifier

	country_modifier_happy = {
		technology_cost = -0.1
		tolerance_heathen = 2
	}
	country_modifier_neutral = {
		tolerance_heathen = 2
	}	
	country_modifier_angry = {
		technology_cost = 0.1
		tolerance_heathen = -1
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {
		local_tax_modifier = 0.33
		local_unrest = -2
		local_missionary_strength = -1		
	}
	province_modifier_neutral = {
		local_tax_modifier = 0.33
		local_missionary_strength = -1		
	}		
	province_modifier_angry = {
		local_unrest = 5
		local_missionary_strength = -1		
	}	
	
	# Influence modifiers
	base_influence = 10
    
	influence_modifier = {
		desc = EST_VAL_RULED_BY_RELIGIOUS_MINORITY
		trigger = {
			NOT = { dominant_religion = ROOT }
			any_owned_province = {
				NOT = { religion_group = religion_group_secular }
			}
		}
		influence = 20
	}
    
	influence_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER
		trigger = {
			OR = {
				has_disaster = estate_settlers_disaster
				has_disaster = estate_bureaucrats_disaster
                has_disaster = estate_officers_disaster
                has_disaster = estate_scientists_disaster
                has_disaster = estate_caravanners_disaster
			}
		}	
		influence = -40
	}
	
	
	# Loyalty modifiers
	loyalty_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER_LOY
		trigger = {
			OR = {
				has_disaster = estate_settlers_disaster
				has_disaster = estate_bureaucrats_disaster
                has_disaster = estate_officers_disaster
                has_disaster = estate_scientists_disaster
			}
		}
		loyalty = -20
	}

    # Custom names
    custom_name = {
		desc = estate_adoratores
		trigger = {
			tag = CAE
		}
	}

	# For province assignment
	ai_grant_province = {
		factor = 1	
		modifier = {
			factor = 0.5
			development = 20
		}
		modifier = {
			factor = 2
			NOT = { development = 10 }
		}				
		modifier = {
			factor = 0.75
			has_owner_culture = yes
		}
		modifier = {
			factor = 0.75
			province_trade_power = 5
		}			
		modifier = {
			factor = 0.75
			province_trade_power = 10
		}			
	}	
    
    # Make Investment
    # years: 10
    # trigger:
    #   always
    # effect:
    #   lose years of income -0.1
    #   add 15 loyalty
    interaction = {
		name = "MAKE_INVESTMENT"
		icon = 6
		
		years_between_use = 10
		
		potential = {
			NOT = { has_disaster = estate_faith_disaster }
		}
		
		trigger = {
			always = yes
		}
		
		effect = {
			add_years_of_income = -0.25
			add_estate_loyalty = {
				estate = estate_faith
				loyalty = 15
			}
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { years_of_income = 0.5 }				
			}
			modifier = {
				factor = 0
				estate_loyalty = {
					estate = estate_faith
					loyalty = 40
				}
			}			
		}		
	}
    
    # Seek Support
    # years: 20
    # trigger:
    #   always
    # effect:
    #   lose prestige/government points
    #   add 10 influence
    #   add 15 loyalty
    interaction = {
		name = "SEEK_SUPPORT"
		icon = 7
		
		years_between_use = 20
		
		trigger = {
			always = yes
		}
		
		effect = {
			add_prestige = -5
			add_legitimacy = -5
			add_republican_tradition = -3
			add_devotion = -3

			add_estate_influence_modifier = {
				estate = estate_faith
				desc = EST_VAL_SOUGHT_SUPPORT
				influence = 10
				duration = 7300
			}
			add_estate_loyalty = {
				estate = estate_faith
				loyalty = 15
			}
		}
	
		ai_will_do = {
			factor = 1
            
			modifier = {
				factor = 0
				NOT = {
                    estate_loyalty = {
                        estate = estate_settlers
                        loyalty = 30
                    }			
                }
			}
			modifier = {
				factor = 0
				estate_influence = {
					estate = estate_settlers
					influence = 40
				}						
			}			
		}
	}
    
    # Oust Undesirables
    # years: 20
    # trigger:
    #   always
    # effect:
    #   -20 influence
    #   -15 loyalty
    interaction = {
		name = "OUST_UNDESIRABLES"
		icon = 10
		
		years_between_use = 20
		
		trigger = {
			always = yes
		}
		
		effect = {
			add_estate_influence_modifier = {
				estate = estate_faith
				desc = EST_VAL_OUSTED_UNDESIRABLES
				influence = -20
				duration = 7300
			}
			add_estate_loyalty = {
				estate = estate_faith
				loyalty = -15
			}
		}
	
		ai_will_do = {
			factor = 1

			modifier = {
				factor = 0
				NOT = {
                    estate_influence = {
                        estate = estate_faith
                        influence = 80
                    }	
                }
			}			
		}
	}
    
    # Recruit Theologian
    # years: 30
    # trigger:
    #   influence: 40
    # effect:
    #   + advisor
    #   15 influence
    interaction = {
		name = "RECRUIT_THEOLOGIAN"
		icon = 25
		
		years_between_use = 30
		
		potential = {
			NOT = { has_disaster = estate_faith_disaster }
		}
		
		trigger = {
			estate_influence = {
				estate = estate_faith
				influence = 40
			}
		}
		
		effect = {
			if = {
				limit = {
					monthly_income = 25
				}
				define_advisor = {
					type = theologian
					skill = 3
					discount = yes
				}
			}
			if = {
				limit = { NOT = { monthly_income = 25 } }
				define_advisor = {
					type = theologian
					skill = 2
					discount = yes
				}
			}
			add_estate_influence_modifier = {
				estate = estate_faith
				desc = EST_VAL_RECRUITED_ADVISOR
				influence = 15
				duration = 10950
			}
		}
		
		ai_will_do = {
			factor = 0
		}
	}
    
    # Recruit Priest
    # years: 30
    # trigger:
    #   influence: 40
    # effect:
    #   + advisor
    #   15 influence
    interaction = {
		name = "RECRUIT_PRIEST"
		icon = 25
		
		years_between_use = 30
		
		potential = {
			NOT = { has_disaster = estate_faith_disaster }
		}
		
		trigger = {
			estate_influence = {
				estate = estate_faith
				influence = 40
			}
		}
		
		effect = {
			if = {
				limit = {
					monthly_income = 25
				}
				define_advisor = {
					type = inquisitor
					skill = 3
					discount = yes
				}
			}
			if = {
				limit = { NOT = { monthly_income = 25 } }
				define_advisor = {
					type = inquisitor
					skill = 2
					discount = yes
				}
			}
			add_estate_influence_modifier = {
				estate = estate_faith
				desc = EST_VAL_RECRUITED_ADVISOR
				influence = 15
				duration = 10950
			}
		}
		
		ai_will_do = {
			factor = 0
		}
	}
 
    # Discover Lost Teachings
    interaction = {
		name = "DISCOVER_LOST_TEXTS"
		icon = 8
		
		years_between_use = 5
		
		potential = {
			NOT = { has_disaster = estate_faith_disaster }
		}
		
		trigger = {
            years_of_income = 1
		}
		
		effect = {
			add_estate_loyalty = {
				estate = estate_faith
				loyalty = 10
			}
            
            add_years_of_income = -1
            
            set_country_flag = search_for_lost_texts
            custom_tooltip = tt_lost_texts
            #country_event = { id = LOST_RELIGIONS.1 }
		}
		
		ai_will_do = {
			factor = 0
		}
	}
