default_scientific_capacity = {
    custom_tooltip = RESET_SCIENTIFIC_CAPACITY
    
    hidden_effect = {
        set_variable = { which = scientific_capacity value = 0 }
    }
}

add_tech_level = {
    custom_tooltip = ADD_TECH_LEVEL
    
    hidden_effect = {
        if = { 
            limit = { NOT = { check_variable = { which = tech_level value = 10 } } }
            change_variable = {
                which = tech_level
                value = 1
            }
        }
    }
}

lose_tech_level = {
    custom_tooltip = LOSE_TECH_LEVEL
    
    hidden_effect = {
        if = { 
            limit = { check_variable = { which = tech_level value = 1 } }
            subtract_variable = {
                which = tech_level
                value = 1
            }
        }
    }
}

add_scientific_capacity_5 = {
    custom_tooltip = GAIN_5_SCIENTIFIC_CAPACITY
    
    hidden_effect = {
        # Set to 100 if above 95
        if = { 
            limit = { check_variable = { which = scientific_capacity value = 95 } }
            set_variable = { which = scientific_capacity value = 100 }
            add_adm_power = 50
        }
           
        # Otherwise add 5
        if = { 
            limit = { NOT = { check_variable = { which = scientific_capacity value = 95 } } }
            change_variable = { which = scientific_capacity value = 5 }
        }
    }
}

lose_scientific_capacity_5 = {
    custom_tooltip = LOSE_5_SCIENTIFIC_CAPACITY
    
    hidden_effect = {
        # Set to 0 if below 5
        if = { 
            limit = { NOT = { check_variable = { which = scientific_capacity value = 5 } } }
            set_variable = { which = scientific_capacity value = 0 }
            add_adm_power = -50
        }
           
        # Otherwise add 5
        if = { 
            limit = { check_variable = { which = scientific_capacity value = 5 } }
            subtract_variable = { which = scientific_capacity value = 5 }
        }
    }
}
