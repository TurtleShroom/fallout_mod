defined_text = {
    name = GetSirOrMadam
    
    text = {
        localisation_key = madam
        trigger = {
			is_female = yes
        }
    }
    
    text = {
        localisation_key = sir
        trigger = {
			NOT = { is_female = yes }
        }
    }
}
