# Random
# Building Missions
build_a_museum = {
	type = our_provinces
	category = ADM
	ai_mission = yes
	
	allow = {
		NOT = { has_building = museum }
		is_city = yes
	}
	abort = {
		always = no
	}
	success = {
		has_building = museum
	}
	chance = {
		factor = 1000
        
        modifier = {
            factor = 0.1
            owner = { NOT = { treasury = 250 } }
        }
        
        modifier = {
            factor = 0.5
            owner = { NOT = { museum = 3 } }
        }
	}
	effect = {
		add_scaled_local_adm_power = 2
	}
}
