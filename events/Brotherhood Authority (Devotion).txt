	title = brotherhood_authority.1.title
	desc = brotherhood_authority.1.desc
    picture = BOS_WORKSHOP_eventPicture
	
    is_triggered_only = yes
    
	trigger = {
        OR = {
            government = administrative_brotherhood_order

            government = fascist_brotherhood_order
        }
	}

    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - brotherhood_authority.1" }
