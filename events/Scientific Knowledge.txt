namespace = scientific_knowledge

# Default technology levels
country_event = {
	id = scientific_knowledge.1
	title = scientific_knowledge.1.title
	desc = scientific_knowledge.1.desc
	picture = BIG_MT_THE_SINK_eventPicture
	
	is_triggered_only = yes
    hidden = yes
    
    trigger = {
        NOT = { has_country_flag = setup_scientific_knowledge_level }
    }
    
    immediate = {
        set_country_flag = setup_scientific_knowledge_level
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - scientific_knowledge.1" }
        }
    }
    
    option = {
		name = scientific_knowledge.1.option.a
        
        set_variable = {
            which = research_progress
            value = 0
        }
        
        set_variable = {
            which = scientific_knowledge_level
            value = 0
        }
        
        if = {
            limit = {
                OR = {
                    technology_group = tech_robot
                    technology_group = tech_commonwealth
                    technology_group = tech_enclave
                    technology_group = tech_shi
                    technology_group = tech_alien
                }
            }
            set_scientific_knowledge_level = { LEVEL = 8 }
            else = {
                if = {
                    limit = {
                        OR = {
                            technology_group = tech_bos
                            technology_group = tech_new_vegas
                        }
                    }
                    set_scientific_knowledge_level = { LEVEL = 6 }
                    else = {
                        if = {
                            limit = {
                                OR = {
                                    technology_group = tech_ncr
                                    technology_group = tech_vault_dweller
                                }
                            }
                            set_scientific_knowledge_level = { LEVEL = 5 }
                            else = {
                                if = {
                                    limit = {
                                        OR = {
                                            technology_group = tech_wastelander
                                            technology_group = tech_ghoul
                                            technology_group = tech_caesars_legion
                                            technology_group = tech_hubologist 
                                        }
                                    }
                                    set_scientific_knowledge_level = { LEVEL = 3 }
                                    else = {
                                        set_scientific_knowledge_level = { LEVEL = 0 }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
	}
}
