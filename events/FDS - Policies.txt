# Economy
country_event = {
	id = fds_policies.1
	title = fds_policies.1.title
	desc = fds_policies.1.desc
    picture = BOS_CHAT_eventPicture
	
    is_triggered_only = yes

	option = {
		name = fds_policies.1.option.a
		if = {
			limit = {
				NOT = { has_country_flag = fds_economy_set_flag }

		if = {
			limit = {
