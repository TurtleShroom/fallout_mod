namespace = province_status

# Township of [Root.GetName]
province_event = {
	id = province_status.1
	title = "province_status.1.title"
	desc = "province_status.1.desc"
	picture = DIAMOND_CITY_eventPicture
    
	trigger = { 
        NOT = { has_province_modifier = pf_town }
        NOT = { has_province_modifier = pf_city }
        NOT = { has_province_modifier = pf_metropolis }
        development = 15
	}
	
    mean_time_to_happen = { months = 120 }
    
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.Owner.GetName] - [Root.GetName] - province_status.1" }
            
            add_province_modifier = {
                name = pf_town
                duration = -1
            }
        }
    }
    
	option = {
		name = province_status.1.option.a
        
        tooltip = {
            add_province_modifier = {
                name = pf_town
                duration = -1
            }
        }
	}	
}
