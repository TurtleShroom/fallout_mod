namespace = research_capacity

# Default technology levels
country_event = {
	id = research_capacity.1
	title = research_capacity.1.title
	desc = research_capacity.1.desc
	picture = BIG_MT_THE_SINK_eventPicture
	
	is_triggered_only = yes
    hidden = yes
    
    trigger = {
        NOT = { has_country_flag = setup_tech }
    }
    
    immediate = {
        set_country_flag = setup_tech
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - research_capacity.1" }
        }
    }
    
    option = {
		name = research_capacity.1.option.a
        
        set_variable = {
            which = scientific_capacity
            value = 0
        }
        
        if = {
            limit = {
                OR = {
                    technology_group = tech_robot
                    technology_group = tech_commonwealth
                    technology_group = tech_enclave
                    technology_group = tech_shi
                    technology_group = tech_alien
                }
            }
            set_variable = {
                which = tech_level
                value = 8
            }
            else = {
                if = {
                    limit = {
                        OR = {
                            technology_group = tech_bos
                            technology_group = tech_new_vegas
                        }
                    }
                    set_variable = {
                        which = tech_level
                        value = 6
                    }
                    else = {
                        if = {
                            limit = {
                                OR = {
                                    technology_group = tech_ncr
                                    technology_group = tech_vault_dweller
                                }
                            }
                            set_variable = {
                                which = tech_level
                                value = 5
                            }
                            else = {
                                if = {
                                    limit = {
                                        OR = {
                                            technology_group = tech_wastelander
                                            technology_group = tech_ghoul
                                            technology_group = tech_caesars_legion
                                            technology_group = tech_hubologist 
                                        }
                                    }
                                    set_variable = {
                                        which = tech_level
                                        value = 3
                                    }
                                    else = {
                                        set_variable = {
                                            which = tech_level
                                            value = 0
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
	}
}
